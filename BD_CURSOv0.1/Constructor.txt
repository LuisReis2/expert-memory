CREATE TABLE users (

    user_name VARCHAR(50) UNIQUE NOT NULL, 
    user_email VARCHAR(100) PRIMARY KEY,
    user_password VARCHAR(50),
    user_type ENUM('ADM', 'STUDENT', 'PROFESSOR')
);

CREATE TABLE CURSOS(
    ID_CURSO INT AUTO_INCREMENT PRIMARY KEY,
    NOME_CURSO VARCHAR(70) NOT NULL,
    DESC_CURSO TEXT NOT NULL
);


CREATE TABLE AULA(
ID_AULA INT AUTO_INCREMENT PRIMARY KEY,
TITLE_AULA VARCHAR(70) NOT NULL,
CONTENT_AULA VARCHAR(280) NOT NULL,
ID_CURSO_FK INT,
FOREIGN KEY (ID_curso_FK) REFERENCES cursos (ID_curso)
);

CREATE TABLE PARTICIPANTES(
ID_PARTICIPANTES INT AUTO_INCREMENT  PRIMARY KEY,
ID_USUARIOS_FK VARCHAR(50) NOT NULL,
FOREIGN KEY(ID_USUARIOS_FK) REFERENCES users(user_email),
ID_CURSO_FK int  NOT NULL,
FOREIGN KEY(ID_CURSO_FK) REFERENCES CURSOS(ID_CURSO)
);
